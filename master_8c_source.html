41 /*
42  * Given a TWI prescaler value, determines the amount of clock periods necessary to
43  * reach a given frequency.
44  */
45 static inline uint32_t clock_periods_from_prescaler(uint32_t target_speed, uint32_t prescaler) {
46 
47  //This function solves the following equation for TWBR (the Two Wire Bitrate Register value),
48  //while avoiding floating point math. Solutions are truncated to the nearest integer.
49  //
50  // target_speed = F_CPU / (16 + 2 * TWBR *(4 ^ prescaler))
51  //
52  //This is useful for computing the a prescalar / TWBR pair that allows a given clock frequency.
53  return (F_CPU / ((2 << (2 * prescaler)) * target_speed)) - (8 >> (prescaler * 2));
54 }

63 /*
64  * Sets up the TWI hardware interface, preparing the TWI hardware for
65  * communications. Unless the TWI clock settings are adjusted, this
66  * method need only be called once.
67  *
68  * @brief Sets up the TWI hardware interface.
69  * @param twi_clock_Speed The clock speed for the TWI interface, in Hz.
70  * You must ensure that ((F_CPU / TWI_CLOCk_SPEED) - 16)/ 2 is less than 256, or this function won't work!
71  * @return none
72  */
73 void set_up_twi_hardware(uint32_t twi_bitrate)
74 {
75 
76  uint8_t prescaler = -1;
77  uint32_t clock_periods;
78 
79  //Compute an appropriate Two Wire prescaler value, which will
80  //determine the rate of the main TWI clock speed.
81  //
82  //The TWI clock speed is defined as follows:
83  //
84  // F_CPU /(16 + 2 * TWBR *(4 ^ prescaler))
85  //
86  // Due to hardware register sizes, two condition have to be met:
87  // 1) The prescaler has to be less than four, as to fit in a two-bit register; and
88  // 2) The "constant" TWBR must be between 0-255, inclusive.
89  //
90  // We'll iterate through prescaler values until we find one that allows us to satisfy condition #2.
91  //
92  do {
93  prescaler += 1;
94  clock_periods = clock_periods_from_prescaler(twi_bitrate, prescaler);
95  }
96  while(clock_periods > 255 && prescaler < 3);
97 
98  //Apply the prescaler we've found.
99  TWSR &= ~0x03;
100  TWSR |= prescaler;
101 
102  //... and apply the Two Wire Bitrate Register (TWBR) value that we've determined.
103  TWBR = clock_periods;
104 
105 }

131 /*
132  * Sends a TWI start condition.
133  *
134  * @retval 0 Returned if we can't communicate with the given device; e.g. if the device doesn't acknowledge communications.
135  * @retval 1 Returned on success.
136  */
137 uint8_t send_twi_start_condition() {
138 
139  //Start the TWI connection.
140  // The following Two Wire Control Register bits are set:
141  // TWEN: Sets the Two Wire ENable bit, which must be written to start any TWI communication.
142  // TWSTA: Sets the Two Wire STArt bit, which specifies that we want to create a start condition.
143  // TWINT: Clears any existing Two Wire INTerrupts, allowing us to move forwad.
144  // Confusingly enough, writing a '1' to this bit clears it.
145  TWCR = (1 << TWINT) | (1 << TWSTA) | (1 << TWEN);
146 
147  wait_for_twi_operation_to_complete();
148 
149  //Check to see if we were succesfully able to gain control of the bus,
150  //indicated by the most significant five bits of the TW_STATUS register.
151  uint8_t two_wire_status = TW_STATUS & 0xF8;
152  return (two_wire_status == TW_START) || (two_wire_status == TW_REP_START);
153 }

157  * Begins an TWI communication to a given address in either read or write mode.
158  * Sends a start bit, the target address, and a driection bit.
159  *
160  * You may prefer to use start_twi_read_from / start_twi_write_to.
161  *
162  * @param address The device's TWI address.
163  * @param direction The communication direction for the given TWI device. Should be either TWI_READ or TWI_WRITE.
164  * @retval 0 Returned if we can't communicate with the given device; e.g. if the device doesn't acknowledge communications.
165  * @retval 1 Returned on success.
166  */
167 uint8_t start_twi_communication(uint8_t address, TWIDataDirection direction)
168 {
169  uint8_t twi_status;
170 
171  //Attempt to send a start condition.
172  //If we fail to take control of the bus, return false.
173  if(!send_twi_start_condition()) {
174  return false;
175  }

181  //or if the status wwas Master Read, SLAve ACKnowledged.
182  return (twi_status == TW_MT_SLA_ACK) || (twi_status == TW_MR_SLA_ACK);

192 static uint8_t raw_twi_write(uint8_t data) {
193 
194  //Send the device's address and direction by populating the
195  //Two Wire Data Register.
196  TWDR = data;
197 
198  //Initiate the actual transmission.
199  // The following Two Wire Control Register bits are set:
200  // TWEN: Sets the Two Wire ENable bit, which must be written to start any TWI communication.
201  // TWINT: Clears any existing Two Wire INTerrupts, allowing us to move forwad.
202  // Confusingly enough, writing a '1' to this bit clears it.
203  TWCR = (1 << TWINT) | (1 << TWEN);
204 
205  wait_for_twi_operation_to_complete();
206 
207  // check value of TWI Status Register. Mask prescaler bits.
208  return TW_STATUS & 0xF8;

223 /*
224  * Attempts to start an TWI communication. If the device responds that it's
225  * not available, retry until the device /is/ available.
226  *
227  * Use of this function is only appropriate in some limited circumstances,
228  * but it is included as to be a complete implementation of the Master TWI library.
229  */
230 void ensure_twi_communication(uint8_t address, TWIDataDirection direction)
231 {
232  uint8_t twi_status;
233 
234  //Repeat until we're able to read succesfully.
235  while(1)
236  {
237 
238  //If we weren't able to start a TWI communication, retry.
239  if (!send_twi_start_condition()) {
240  continue;
241  }
242 
243  //Send the device's address, and the direction bit.
244  twi_status = raw_twi_write(address << 1 | direction);
245 
246  //If we recieved a Negative ACKnoledgement of either type
247  //(Master Transmit, SLAve Negative ACKnowledgement OR
248  // Master Receive, SLAve Negative ACKnowledgement),
249  // release the bus, and wait for the device to be ready.
250  if((twi_status == TW_MT_SLA_NACK ) || (twi_status == TW_MR_DATA_NACK)) {
251  //Abort the current TWI packet, and wait for the device to report that it's ready.
252  end_twi_packet();
253  }
254  //Otherwise, mark the communication as complete, and stop.
255  else {
256  break;
257  }
258  }

262 /*
263  * Terminates TWI communication with the given bus.
264  */
265 void end_twi_packet()
266 {
267  //Terminate the active TWI packet by sending a stop condition.
268  // The following Two Wire Control Register bits are set:
269  // TWEN: Sets the Two Wire ENable bit, which must be written to start any TWI communication.
270  // TWSTO: Sets the Two Wire STOp bit, which specifies that we want to terminate our TWI communication.
271  // TWINT: Clears any existing Two Wire INTerrupts, allowing us to move forwad.
272  // Confusingly enough, writing a '1' to this bit clears it.
273  TWCR = (1 << TWINT) | (1 << TWSTO) | (1 << TWEN);
274 
275  //Wait until we're no longer sending a Two Wire STOp condition.
276  while(TWCR & (1 << TWSTO));
277 }

296 /*
297  * Reads a single byte via TWI, and the requests more.
298  *
299  * @param read_mode Specifies the read mode for the given TWI communication,
300  * which in turn specifies whether an additional byte is requested.
301  * @return The byte read via TWI.
302  */
303 uint8_t read_via_twi(TWIReadMode read_mode)
304 {
305  //Ininitate a TWI read.
306  // The following Two Wire Control Register bits are set:
307  // TWEN: Sets the Two Wire ENable bit, which must be written to start any TWI communication.
308  // TWINT: Clears any existing Two Wire INTerrupts, allowing us to move forwad.
309  // Confusingly enough, writing a '1' to this bit clears it.
310  // TWEA: Sets the Two Wire Enable Acknowledge bit, which indicates that we're expecting more data.
311  //
312  //The lack of any other instruction (e.g. TWSTA) indicates that we're reading.
313  TWCR = (1 << TWINT) | (1 << TWEN) | (read_mode << TWEA);
314  wait_for_twi_operation_to_complete();
315 
316  return TWDR;
317 }

319 /*
320  * Performs a given bus pirate command.
321  *
322  * Supports the following features:
323  * {}, [], R/r, 0-255, 0b, 0h, &
324  *
325  * See: http://dangerousprototypes.com/bus-pirate-manual/i2c-guide/
326  *
327  * Two additional commands are also implemented:
328  * s: Reads a single byte, and then responds with a NAK.
329  * w: Transmits a single byte, provided as an argument. This allows programatic
330  * control of transmission.
331  *
332  * Important note! You'll need to replace your last "r" with an "s",
333  * or the TWI library will "lock up", as the AVR views the transmission
334  * as being incomplete.
335  *
336  * For each read, a pointer should be provided to a uint8_t target.
337  * For example:
338  *
339  * @code
340  * uint8_t hello;
341  * perform_bus_pirate_twi_command("[ 0x72 0x80 0x03 [ 0x73 s ]", &hello);
342  * @endcode
343  *
344  * would read a single byte to the variable hello.
345  *
346  * For each _write_, a (non-pointer) uint8_t should be provided.
347  * For example:
348  *
349  * @code
350  * perform_bus_pirate_twi_command("[ 0x72 0x80 w ]", 0x55);
351  * @endcode
352  *
353  * would send 0x72 (the address and write bit), then 0x80, and then 0x55.
354  *
355  *
356  * @param command The bus pirate command, as a null-terminated string.
357  * @param ... A single uint8_t * for each read command, or a single uint8_t for each write.
358  *
359  */
360 uint8_t perform_bus_pirate_twi_command(const char * command, ...) {
361 
362 
363  //Stores the current radix, which is decimal by default.
364  uint8_t radix = 10;
365  uint8_t to_transmit = 0, is_transmission = 0, read_count = 0;
366 
367  //Set up handling of the variadic arguments, which are used
368  //for reading.
369  va_list variadic_arguments;
370  uint8_t * read_target = 0;
371  va_start(variadic_arguments, command);
372 
373  //Process each argument in the command.
374  for(; *command; ++command) {
375  switch(*command) {
376 
377  //If we have an open brace, issue a TWI start condition.
378  case '{':
379  case '[':
380  send_twi_start_condition();
381  break;
382 
383  //If we have a close brace, issue a TWI stop condition.
384  case '}':
385  case ']':
386  end_twi_packet();
387  break;
388 
389  //If we have an 'x', switch the active radix to hex.
390  case 'x':
391  radix = 16;
392  break;
393 
394  //If we have a 'b', switch the active radix to binary.
395  case 'b':
396  radix = 2;
397  break;
398 
399 
400  //Read a single byte from the TWI device, and then
401  //send an acknowledgement, indicating that we expect further data.
402  case 'r':
403  case 'R':
404  case 's':
405  case 'S':
406  {
407 
408  TWIReadMode read_mode;
409 
410  //If we're been provided the 's' command,
411  //set the read mode to "last byte", which sends
412  //a negative acknowledgement to the device.
413  if(*command == 's' || *command == 'S') {
414  read_mode = LastByte;
415  }
416  //Otherwise, send an ACK, indiacting that we
417  //would like additional data.
418  else {
419  read_mode = RequestMore;
420  }
421 
422  //Get the address that the read is targeting...
423  read_target = va_arg(variadic_arguments, uint8_t *);
424 
425  //... and then perform the read.
426  *read_target = read_via_twi(read_mode);
427 
428  //... increment the number of total reads, and continue.
429  ++read_count;
430  break;
431  }
432 
433  // Send a single byte via the TWI interface, which should be
434  // provided as an argument.
435  case 'w':
436  case 'W':
437 
438  //Since we're writing, this is a transmission.
439  is_transmission = 1;
440 
441  //Read the byte that should be transmitted...
442  to_transmit = va_arg(variadic_arguments, unsigned int);
443 
444  //... and "roll" directly into the transmit case.
445 
446  //If we have a delimiter, handle any actions that have been queued.
447  case ' ':
448  case ',':
449  if(is_transmission) {
450  send_via_twi(to_transmit);
451  }
452 
453  //Reset our state to the default.
454  radix = 10;
455  is_transmission = to_transmit = 0;
456  break;
457 
458  //Delay 1us.
459  case '&':
460  _delay_us(1);
461  break;
462 
463  //In all other cases, check to see if we have a piece of a literal.
464  default:
465  {
466 
467  uint8_t numeric_value;
468  uint8_t is_lowercase_hex = (*command >= 'a') && (*command <= 'f') && (radix == 16);
469  uint8_t is_uppercase_hex = (*command >= 'A') && (*command <= 'F') && (radix == 16);
470  uint8_t is_decimal_digit = (*command >= '0') && (*command <= '9') && (radix >= 10);
471  uint8_t is_valid_binary = (*command >= '0') && (*command <= '1') && (radix == 2);
472 
473  //If we have a number format, convert the number to a raw numeric value.
474  if(is_lowercase_hex) {
475  numeric_value = *command - 'a' + 10;
476  }
477  else if(is_uppercase_hex) {
478  numeric_value = *command - 'A' + 10;
479  }
480  else if(is_decimal_digit || is_valid_binary) {
481  numeric_value = *command - '0';
482  }
483  //Otherwise, the value is nonsensical, and should be skipped.
484  else {
485  break;
486  }
487 
488  //Since we have a valid value to transmit, mark this as a transmission.
489  //This is idempotent, so it doesn't matter if this is run multiple times.
490  is_transmission = 1;
491 
492  //Move the existing number over by a single radix place, "making room"
493  //to add the new number to the right...
494  to_transmit *= radix;
495 
496  //... and then add the new value in the vacated spot.
497  to_transmit += numeric_value;
498 
499  break;
500  }
501  }
502  }
503 
504  //Halt parsing of varaidic arguments.
505  va_end(variadic_arguments);
506 
507  //Return the number of reads performed.
508  return read_count;
509 }
